VBA Script to Save Active Sheet as PDF

The Script

This macro saves the current sheet as a PDF in the same folder as the Excel file.

Sub SavePDF()
    Dim PDFName As String
    
    If ThisWorkbook.Path = "" Then
        MsgBox "Save file first!"
        Exit Sub
    End If
    
    PDFName = ThisWorkbook.Path & "\" & ActiveSheet.Name & ".pdf"
    
    ActiveSheet.ExportAsFixedFormat Type:=xlTypePDF, Filename:=PDFName
    
    MsgBox "PDF Saved: " & PDFName
End Sub

How to Merge Multiple Excel Files into One (VBA Script)

The Script

This script loops through a folder and combines all Excel files into your current workbook.

Sub MergeFiles()
    Dim FolderPath As String, Filename As String
    Dim Sheet As Worksheet, LastRow As Long
    Dim Master As Workbook, Source As Workbook
    
    ' CHANGE THIS PATH (Keep the slash at the end)
    FolderPath = "C:\Users\YourName\Documents\Files\"
    
    Set Master = ThisWorkbook
    Filename = Dir(FolderPath & "*.xlsx")
    
    Do While Filename <> ""
        Set Source = Workbooks.Open(FolderPath & Filename)
        LastRow = Source.Sheets(1).Cells(Rows.Count, 1).End(xlUp).Row
        
        If LastRow > 1 Then
            Source.Sheets(1).Range("A2:Z" & LastRow).Copy
            Master.Sheets(1).Range("A" & Rows.Count).End(xlUp).Offset(1, 0).PasteSpecial xlPasteValues
        End If
        
        Source.Close False
        Filename = Dir
    Loop
    MsgBox "Done!"
End Sub
